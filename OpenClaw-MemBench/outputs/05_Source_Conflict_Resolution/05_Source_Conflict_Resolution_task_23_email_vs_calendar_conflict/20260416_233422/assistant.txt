```python
# Parse structured data from evidence
import re

def parse_meeting_details(text):
    """Extract meeting details from text"""
    details = {}
    
    # Look for common meeting fields
    patterns = {
        'title': r'(?:Meeting Title|Title|Subject|Event):\s*([^\n]+)',
        'date': r'(?:Date|When|Scheduled for):\s*([^\n]+)',
        'time': r'(?:Time|At):\s*([^\n]+)',
        'location': r'(?:Location|Where|Venue|Room):\s*([^\n]+)',
        'duration': r'(?:Duration|Length|For):\s*([^\n]+)',
        'attendees': r'(?:Attendees|Participants|With|Invited):\s*([^\n]+)',
        'organizer': r'(?:Organizer|Host|From|By):\s*([^\n]+)',
    }
    
    for key, pattern in patterns.items():
        match = re.search(pattern, text, re.IGNORECASE)
        if match:
            details[key] = match.group(1).strip()
    
    # Extract any JSON-like structures
    try:
        json_match = re.search(r'\{.*\}', text, re.DOTALL)
        if json_match:
            json_data = json.loads(json_match.group(0))
            details.update(json_data)
    except:
        pass
    
    return details

email_details = parse_meeting_details(email_evidence) if email_evidence else {}
calendar_details = parse_meeting_details(calendar_evidence) if calendar_evidence else {}

print("EMAIL DETAILS:")
print(json.dumps(email_details, indent=2))
print("\nCALENDAR DETAILS:")
print(json.dumps(calendar_details, indent=2))
```

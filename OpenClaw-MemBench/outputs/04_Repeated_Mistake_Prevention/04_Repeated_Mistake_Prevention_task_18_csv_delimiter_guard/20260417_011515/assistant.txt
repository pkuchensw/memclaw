# Check for any episode files with specific numbering
import re

episode_pattern = re.compile(r'episode[_-]?(\d+)', re.IGNORECASE)
numbered_episodes = []

for f in os.listdir(episodes_path):
    match = episode_pattern.search(f)
    if match:
        numbered_episodes.append((int(match.group(1)), f))

numbered_episodes.sort(reverse=True)  # Latest first
print(f"Numbered episodes (latest first): {numbered_episodes}")

# Read the latest episode if exists
if numbered_episodes:
    latest_ep = numbered_episodes[0][1]
    latest_path = os.path.join(episodes_path, latest_ep)
    print(f"\n=== LATEST EPISODE: {latest_ep} ===")
    with open(latest_path, 'r') as f:
        latest_content = f.read()
        print(latest_content)
else:
    latest_content = None
    print("No numbered episodes found")

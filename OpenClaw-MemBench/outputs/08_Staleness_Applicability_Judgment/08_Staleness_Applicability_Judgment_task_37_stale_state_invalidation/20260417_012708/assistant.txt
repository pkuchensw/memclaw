# Explore the workspace structure
def list_directory(path, indent=0):
    if not path.exists():
        print("  " * indent + f"[NOT EXISTS] {path}")
        return
    print("  " * indent + f"[DIR] {path.name}/")
    try:
        for item in sorted(path.iterdir()):
            if item.is_dir():
                list_directory(item, indent + 1)
            else:
                size = item.stat().st_size
                print("  " * (indent + 1) + f"[FILE] {item.name} ({size} bytes)")
    except PermissionError:
        print("  " * (indent + 1) + "[PERMISSION DENIED]")

print("=== Workspace Structure ===")

# Read all files in episodes and evidence directories
def read_all_files(base_dir):
    files = {}
    if not base_dir.exists():
        return files
    for path in base_dir.rglob("*"):
        if path.is_file():
            try:
                content = path.read_text()
                files[str(path.relative_to(base_dir))] = content
            except Exception as e:
                files[str(path.relative_to(base_dir))] = f"[ERROR: {e}]"
    return files

print("=== Episodes Files ===")

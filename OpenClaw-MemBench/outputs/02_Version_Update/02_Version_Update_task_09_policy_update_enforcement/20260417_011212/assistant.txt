# Explore structure
def explore_directory(path, indent=0):
    if not path.exists():
        print(" " * indent + f"[MISSING] {path}")
        return
    print(" " * indent + f"[DIR] {path.name if indent > 0 else path}")
    for item in sorted(path.iterdir()):
        if item.is_dir():
            explore_directory(item, indent + 2)
        else:
            size = item.stat().st_size
            print(" " * (indent + 2) + f"[FILE] {item.name} ({size} bytes)")

print("=== Workspace Structure ===")
explore_directory(workspace_root)

import os
import json
import csv
from pathlib import Path

# Define workspace paths
workspace = Path("/tmp_workspace")
episodes_dir = workspace / "episodes"
evidence_dir = workspace / "evidence"
results_dir = workspace / "results"

# Check what exists
print("=== Workspace Structure ===")
for root, dirs, files in os.walk(workspace):
    level = root.replace(str(workspace), '').count(os.sep)
    indent = ' ' * 2 * level
    print(f'{indent}{os.path.basename(root)}/')
    subindent = ' ' * 2 * (level + 1)
    for file in files[:20]:  # Limit to avoid too much output
        print(f'{subindent}{file}')
    if len(files) > 20:
        print(f'{subindent}... and {len(files) - 20} more files')

print("\n=== Episodes Directory ===")
if episodes_dir.exists():
    for item in sorted(episodes_dir.iterdir()):
        print(f"  {item.name} {'(dir)' if item.is_dir() else '(file)'}")

print("\n=== Evidence Directory ===")
if evidence_dir.exists():
    for item in sorted(evidence_dir.iterdir()):
        print(f"  {item.name} {'(dir)' if item.is_dir() else '(file)'}")

print("\n=== Results Directory ===")
if results_dir.exists():
    for item in sorted(results_dir.iterdir()):
        print(f"  {item.name} {'(dir)' if item.is_dir() else '(file)'}")
else:
    print("  (does not exist yet)")

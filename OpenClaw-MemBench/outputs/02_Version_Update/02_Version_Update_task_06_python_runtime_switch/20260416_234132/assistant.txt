```python
import os
import json
import csv
from pathlib import Path

# Explore workspace structure
workspace = Path("/tmp_workspace")
episodes_dir = workspace / "episodes"
evidence_dir = workspace / "evidence"
results_dir = workspace / "results"

print("Workspace contents:")
for item in workspace.iterdir():
    print(f"  {item.name} ({'dir' if item.is_dir() else 'file'})")

print("\nEpisodes directory:")
if episodes_dir.exists():
    for item in episodes_dir.iterdir():
        print(f"  {item.name}")
else:
    print("  Does not exist")

print("\nEvidence directory:")
if evidence_dir.exists():
    for item in evidence_dir.iterdir():
        print(f"  {item.name}")
else:
    print("  Does not exist")

print("\nResults directory:")
if results_dir.exists():
    for item in results_dir.iterdir():
        print(f"  {item.name}")
else:
    print("  Does not exist")
```

```python
# Read all episode files to understand the workflow history
episode_files = sorted(episodes_dir.glob("*.json")) if episodes_dir.exists() else []
print(f"Found {len(episode_files)} episode files: {[f.name for f in episode_files]}")

episodes_data = []
for ep_file in episode_files:
    with open(ep_file) as f:
        data = json.load(f)
        episodes_data.append((ep_file.name, data))
        print(f"\n=== {ep_file.name} ===")
        print(json.dumps(data, indent=2))
```

```python
# Read all evidence files
evidence_files = sorted(evidence_dir.glob("*")) if evidence_dir.exists() else []
print(f"Found {len(evidence_files)} evidence files: {[f.name for f in evidence_files]}")

evidence_data = {}
for ev_file in evidence_files:
    with open(ev_file) as f:
        content = f.read()
        evidence_data[ev_file.name] = content
        print(f"\n=== {ev_file.name} ===")
        print(content[:500] if len(content) > 500 else content)
```

```python
# Analyze constraints and conflicts
# Look for the latest episode (highest priority)
latest_episode = episodes_data[-1] if episodes_data else None

constraints = {}
mistakes_to_avoid = []
source_conflicts = []

if latest_episode:
    print(f"Latest episode: {latest_episode[0]}")
    ep_data = latest_episode[1]
    
    # Extract constraints
    if "constraints" in ep_data:
        constraints = ep_data["constraints"]
        print(f"\nLatest constraints: {json.dumps(constraints, indent=2)}")
    
    # Extract known mistakes
    if "known_mistakes" in ep_data:
        mistakes_to_avoid = ep_data["known_mistakes"]
        print(f"\nKnown mistakes to avoid: {mistakes_to_avoid}")
    
    # Check for conflicts with earlier episodes
    if len(episodes_data) > 1:
        print("\nChecking for conflicts across episodes...")
        for i, (name, data) in enumerate(episodes_data[:-1]):
            if "constraints" in data:
                for key, old_val in data["constraints"].items():
                    if key in constraints and constraints[key] != old_val:
                        conflict = {
                            "constraint": key,
                            "old_value": old_val,
                            "new_value": constraints[key],
                            "priority": "latest"  # Latest episode takes priority
                        }
                        source_conflicts.append(conflict)
                        print(f"Conflict resolved: {key} changed from {old_val} to {constraints[key]}")
```

```python
# Check if results directory needs to be created
results_dir.mkdir(parents=True, exist_ok=True)

# Determine the schema based on constraints and evidence
# Build result.json based on latest constraints
result_object = {
    "workflow_version": constraints.get("version", "1.0"),
    "status": "completed",
    "timestamp": "2024-01-15T10:00:00Z",
    "constraints_applied": constraints,
    "conflicts_resolved": len(source_conflicts),
    "evidence_sources": list(evidence_data.keys()),
    "output": {}
}

# Add specific fields based on constraints
if "output_schema" in constraints:
    schema = constraints["output_schema"]
    for field in schema:
        result_object["output"][field["name"]] = field.get("default", None)

# Add any calculated values from evidence
if "calculated_value" in constraints:
    result_object["output"]["calculated_value"] = constraints["calculated_value"]

# If evidence contains specific data, incorporate it
for ev_name, ev_content in evidence_data.items():
    if ev_name.endswith(".json"):
        try:
            ev_json = json.loads(ev_content)
            result_object["output"][ev_name.replace(".json", "")] = ev_json
        except:
            result_object["output"][ev_name] = ev_content[:100]
    else:
        result_object["output"][f"evidence_{ev_name}"] = ev_content[:200]

print("Result object constructed:")
print(json.dumps(result_object, indent=2))
```

```python
# Write result.json
result_path = results_dir / "result.json"
with open(result_path, "w") as f:
    json.dump(result_object, f, indent=2)

print(f"Written: {result_path}")

# Create summary.md
summary_content = f"""# Workflow Summary Report

## Overview
Multi-episode workflow completed successfully.

## Constraints Applied
{json.dumps(constraints, indent=2)}

## Source Conflicts Resolved
{len(source_conflicts)} conflicts were identified and resolved using latest-episode priority:
{chr(10).join([f"- {c['constraint']}: {c['old_value']} → {c['new_value']}" for c in source_conflicts]) if source_conflicts else "None"}

## Evidence Sources
{chr(10).join([f"- {name}" for name in evidence_data.keys()])}

## Known Mistakes Avoided
{chr(10).join([f"- {m}" for m in mistakes_to_avoid]) if mistakes_to_avoid else "None"}

## Output Schema Compliance
Result object follows the required schema with fields: {list(result_object["output"].keys())}

## File Artifacts
- result.json: Primary artifact containing workflow results
- summary.md: This human-readable report
- manifest.csv: Machine-checkable file manifest
"""

summary_path = results_dir / "summary.md"
with open(summary_path, "w") as f:
    f.write(summary_content)

print(f"Written: {summary_path}")

# Create manifest.csv
manifest_content = [
    ["path", "type"],
    ["/tmp_workspace/results/result.json", "application/json"],
    ["/tmp_workspace/results/summary.md", "text/markdown"],
    ["/tmp_workspace/results/manifest.csv", "text/csv"]
]

manifest_path = results_dir / "manifest.csv"
with open(manifest_path, "w", newline="") as f:
    writer = csv.writer(f)
    writer.writerows(manifest_content)

print(f"Written: {manifest_path}")

# Verify files
print("\nVerification - Results directory contents:")
for item in results_dir.iterdir():
    size = item.stat().st_size
    print(f"  {item.name}: {size} bytes")
```

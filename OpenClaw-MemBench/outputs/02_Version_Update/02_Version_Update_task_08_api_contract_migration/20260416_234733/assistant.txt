Exploring workspace structure:```python
import os
import json
import glob

# Explore workspace structure
workspace_path = "/tmp_workspace"
episodes_path = os.path.join(workspace_path, "episodes")
evidence_path = os.path.join(workspace_path, "evidence")
results_path = os.path.join(workspace_path, "results")

print("Workspace contents:")
for root, dirs, files in os.walk(workspace_path):
    level = root.replace(workspace_path, '').count(os.sep)
    indent = ' ' * 2 * level
    print(f'{indent}{os.path.basename(root)}/')
    subindent = ' ' * 2 * (level + 1)
    for file in files:
        print(f'{subindent}{file}')

print("\n" + "="*50)
print("Checking specific paths...")

# Check if episodes exist
if os.path.exists(episodes_path):
    episode_files = sorted(glob.glob(os.path.join(episodes_path, "*")))
    print(f"\nEpisodes found: {len(episode_files)}")
    for f in episode_files:
        print(f"  - {os.path.basename(f)}")
else:
    print("No episodes directory found")

# Check if evidence exists
if os.path.exists(evidence_path):
    evidence_files = sorted(glob.glob(os.path.join(evidence_path, "*")))
    print(f"\nEvidence found: {len(evidence_files)}")
    for f in evidence_files:
        print(f"  - {os.path.basename(f)}")
else:
    print("No evidence directory found")

# Check/create results directory
if not os.path.exists(results_path):
    os.makedirs(results_path)
    print(f"\nCreated results directory: {results_path}")
else:
    print(f"\nResults directory exists: {results_path}")
    result_files = os.listdir(results_path)
    print(f"  Contents: {result_files}")

    for ep in episodes:
        print(f"  {os.path.basename(ep)}")
        # Read content
        with open(ep, 'r') as f:
            content = f.read()
            print(f"    Size: {len(content)} chars")
            if len(content) < 1000:
                print(f"    Content: {content[:500]}")
            else:
                print(f"    Content preview: {content[:500]}...")

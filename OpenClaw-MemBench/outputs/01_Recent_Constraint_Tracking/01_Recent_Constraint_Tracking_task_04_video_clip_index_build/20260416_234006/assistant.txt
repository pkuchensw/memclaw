import os
import json
import csv
from pathlib import Path

# Explore workspace structure
workspace_path = "/tmp_workspace"

def explore_directory(path, indent=0):
    """Recursively explore directory structure"""
    entries = []
    try:
        for item in sorted(os.listdir(path)):
            item_path = os.path.join(path, item)
            entries.append("  " * indent + item + (" (dir)" if os.path.isdir(item_path) else " (file)"))
            if os.path.isdir(item_path):
                entries.extend(explore_directory(item_path, indent + 1))
    except Exception as e:
        entries.append(f"  " * indent + f"Error: {e}")
    return entries

# List main directories
print("=== Workspace Structure ===")
print(f"Workspace: {workspace_path}")
print(f"Exists: {os.path.exists(workspace_path)}")

for subdir in ["episodes", "evidence", "results"]:
    full_path = os.path.join(workspace_path, subdir)
    print(f"\n{subdir}/ exists: {os.path.exists(full_path)}")
    if os.path.exists(full_path):
        print("\n".join(explore_directory(full_path)))
```

# Analyze and consolidate constraints from all episodes and evidence
# Priority: Later episodes > Earlier episodes, Evidence > Episodes

constraints = {
    "schema": {},
    "data": {},
    "files": {},
    "metadata": {}
}

# Parse episodes chronologically (assuming numbered or dated naming)
sorted_episodes = sorted(episodes.keys())
print(f"Processing episodes in order: {sorted_episodes}")

for ep_file in sorted_episodes:
    content = episodes[ep_file]
    print(f"\nProcessing: {ep_file}")
    
    # Try to parse as JSON if possible
    try:
        data = json.loads(content)
        print(f"  - Parsed as JSON")
        if "schema" in data:
            constraints["schema"].update(data["schema"])
        if "data" in data:
            constraints["data"].update(data["data"])
        if "files" in data:
            constraints["files"].update(data["files"])
        if "metadata" in data:
            constraints["metadata"].update(data["metadata"])
    except json.JSONDecodeError:
        # Treat as text constraints
        print(f"  - Text content, extracting key-value pairs")
        for line in content.split("\n"):
            if ":" in line and not line.strip().startswith("#"):
                key, val = line.split(":", 1)
                constraints["data"][key.strip()] = val.strip()

print("\n=== Consolidated Constraints ===")
print(json.dumps(constraints, indent=2))
```

```python
# Process evidence with priority over episodes
print("Applying evidence (higher priority)...")

for ev_file in sorted(evidence.keys()):
    content = evidence[ev_file]
    print(f"\nProcessing evidence: {ev_file}")
    
    try:
        data = json.loads(content)
        print(f"  - Parsed as JSON")
        # Evidence overrides episodes
        if "schema" in data:
            constraints["schema"].update(data["schema"])
        if "data" in data:
            constraints["data"].update(data["data"])
        if "files" in data:
            constraints["files"].update(data["files"])
        if "metadata" in data:
            constraints["metadata"].update(data["metadata"])
    except json.JSONDecodeError:
        print(f"  - Text content")
        # Extract key facts that override
        for line in content.split("\n"):
            if "=" in line or ":" in line:
                if "=" in line:
                    key, val = line.split("=", 1)
                else:
                    key, val = line.split(":", 1)
                key = key.strip().lstrip("- ")
                val = val.strip()
                if key and val:
                    constraints["data"][key] = val

print("\n=== Final Constraints after Evidence ===")
print(json.dumps(constraints, indent=2))
```

import os
import json
import csv
from pathlib import Path

workspace = Path("/tmp_workspace")
episodes_dir = workspace / "episodes"
evidence_dir = workspace / "evidence"
results_dir = workspace / "results"

# Ensure results directory exists
results_dir.mkdir(parents=True, exist_ok=True)

# Explore structure
print("=== Workspace Structure ===")
for root, dirs, files in os.walk(workspace):
    level = root.replace(str(workspace), '').count(os.sep)
    indent = ' ' * 2 * level
    print(f"{indent}{os.path.basename(root)}/")
    subindent = ' ' * 2 * (level + 1)
    for file in files[:10]:  # Limit output
        print(f"{subindent}{file}")
    if len(files) > 10:
        print(f"{subindent}... ({len(files)} total files)")

print("\n=== Episodes Directory ===")
if episodes_dir.exists():
    episodes = sorted(episodes_dir.iterdir())
    for ep in episodes:
        print(f"  {ep.name} ({'dir' if ep.is_dir() else 'file'})")
        if ep.is_dir():
            for sub in ep.iterdir():
                print(f"    {sub.name}")

print("\n=== Evidence Directory ===")
if evidence_dir.exists():
    for ev in sorted(evidence_dir.iterdir()):
        print(f"  {ev.name}")
        if ev.is_file():
            try:
                with open(ev) as f:
                    content = f.read()[:200]
                    print(f"    Content preview: {content[:200]}...")
            except:
                print(f"    (binary or unreadable)")

# Try to extract any actual constraints from episode files
if os.path.exists(episodes_dir):
    for filename in os.listdir(episodes_dir):
        if filename.endswith('.md'):
            filepath = os.path.join(episodes_dir, filename)
            with open(filepath, 'r') as f:
                content = f.read()
                result_data["episodes_processed"].append({
                    "file": filename,
                    "size": len(content),
                    "processed": True
                })
                # Extract constraint mentions
                if "constraint" in content.lower():
                    result_data["constraints"]["preserved_constraints"].append({
                        "source": filename,
                        "note": "constraint mentioned in episode context"
                    })

# Try to extract evidence sources
if os.path.exists(evidence_dir):
    for filename in os.listdir(evidence_dir):
        if filename.endswith('.md'):
            filepath = os.path.join(evidence_dir, filename)

                content = f.read()
                result_data["evidence_used"].append({
